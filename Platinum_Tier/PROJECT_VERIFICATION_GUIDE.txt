METHOD 8: AUTOMATED VERIFICATION SCRIPT
=================================================================

Create a script that verifies everything automatically:

```python
# verify_system.py
import os
import subprocess
import time
from datetime import datetime

def verify_system():
    print("=" * 60)
    print("AI EMPLOYEE SYSTEM VERIFICATION")
    print("=" * 60)
    print()

    # Check 1: PM2 processes
    print("1. Checking PM2 processes...")
    result = subprocess.run(['pm2', 'list'], capture_output=True, text=True)
    if 'gmail-watcher' in result.stdout and 'online' in result.stdout:
        print("   ✓ Gmail watcher: RUNNING")
    else:
        print("   ✗ Gmail watcher: NOT RUNNING")

    if 'linkedin-watcher' in result.stdout and 'online' in result.stdout:
        print("   ✓ LinkedIn watcher: RUNNING")
    else:
        print("   ✗ LinkedIn watcher: NOT RUNNING")

    if 'filesystem-watcher' in result.stdout and 'online' in result.stdout:
        print("   ✓ Filesystem watcher: RUNNING")
    else:
        print("   ✗ Filesystem watcher: NOT RUNNING")

    # Check 2: File counts
    print("\n2. Checking file activity...")
    inbox_count = len([f for f in os.listdir('Inbox') if os.path.isfile(os.path.join('Inbox', f))])
    done_count = len([f for f in os.listdir('Done') if os.path.isfile(os.path.join('Done', f))])

    print(f"   ✓ Files in Inbox: {inbox_count}")
    print(f"   ✓ Files processed: {done_count}")

    if inbox_count > 0 and done_count > 0:
        print("   ✓ System is processing files")
    else:
        print("   ✗ No file activity detected")

    # Check 3: Create test file and verify processing
    print("\n3. Testing live processing...")
    timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
    test_file = f"Inbox/VERIFY_{timestamp}.txt"

    with open(test_file, 'w') as f:
        f.write("Verification test file")
    print(f"   ✓ Created test file: {test_file}")

    print("   Waiting 5 seconds for processing...")
    time.sleep(5)

    # Check if file was processed
    done_files = os.listdir('Done')
    if any(f"VERIFY_{timestamp}" in f for f in done_files):
        print("   ✓ Test file was processed successfully!")
    else:
        print("   ✗ Test file was NOT processed")

    # Check 4: Dashboard
    print("\n4. Checking Dashboard...")
    if os.path.exists('Dashboard.md'):
        with open('Dashboard.md', 'r') as f:
            content = f.read()
            entry_count = content.count("## File Processed")
        print(f"   ✓ Dashboard exists with {entry_count} entries")
    else:
        print("   ✗ Dashboard not found")

    # Check 5: Logs
    print("\n5. Checking logs...")
    log_files = [
        'Gold_Tier/Logs/gmail_watcher.log',
        'Gold_Tier/Logs/linkedin_watcher.log',
        'Gold_Tier/Logs/filesystem_watcher.log'
    ]

    for log_file in log_files:
        if os.path.exists(log_file):
            print(f"   ✓ {log_file} exists")
        else:
            print(f"   ✗ {log_file} not found")

    print("\n" + "=" * 60)
    print("VERIFICATION COMPLETE")
    print("=" * 60)

if __name__ == "__main__":
    verify_system()
```

Run verification:
```bash
python verify_system.py
```
